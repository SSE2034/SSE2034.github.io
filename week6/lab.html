void matSub(int A[][N], int B[][N], int C[][N]);

C = A - B

Perform subtraction operation between matrix A and B.

And transfer the result to matrix C.

int A[2][2] = {{1,2},{3,4}};
int B[2][2] = {{1,2},{4,8}};
int C[2][2] = {0,};

matSub(A, B, C);

// Now C={{0,0},{-1,-4}}

void matMul(int A[][N], int B[][N], int C[][N]);

C = A.dot(B)

Perform multiplication operation between matrix A and B.

And transfer the result to matrix C.

int A[2][2] = {{1,2},{3,4}};
int B[2][2] = {{1,2},{3,4}};
int C[2][2] = {0,}

matMul(A, B, C);

// Now C = {{7, 10}, {15, 22}}

void transpose(int A[][N], int A_T[][N]);

A_T = A.transpose()

Perform transportation of matrix A.

And transfer the result to matrix A_T.

int A[2][2] = {{1,2},{3,4}};
int A_T[2][2] = {0,}

transpose(A, A_T);

// Now A_T = {{1, 3}, {2, 4}}

string matrixToString(int A[][N]);

Return a string converted from matrix array.

int A[2][2] = {{1,2},{3,4}};

string str = matrixToString(A);

std::cout << str << std::endl;

// {1,2},
// {3,4}
